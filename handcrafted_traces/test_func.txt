// алгоритм alg_fact1 (факториал)
{
	функция fact(int n)
	{
		если условие (n <= 2)  // проверка_базы
			return 1
		иначе
			value = fact(n-1)
			return n * value
	}
	
	функция fact2(int x)
	{
		если условие (x <= 2)  // проверка_базы (в fact2)
			return 1
		иначе
			return fact2(x-1) * x
	}
	
	assert(fact(5) == 120)
	assert(fact2(5) == 120)
}
// alg_fact1 - (0)
// трасса 
{
началась программа
закончилась программа
}
